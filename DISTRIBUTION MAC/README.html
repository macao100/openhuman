Installation express

# 1. Vérifier les prérequis (macOS 14+)
bash prerequis-mac.sh --install

# 2. Installer DADOU (mode hors-ligne avec Ollama)
bash install.sh --offline

# 3. Lancer l'interface desktop
cd ~/dadou
pnpm dev:app
