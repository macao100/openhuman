       ```rust
           engine: Engine,
       }
       
       impl WasmEngine {
           pub fn new() -> Result<Self> {
               let mut config = Config::default();
               config.epoch_interruption(true); // Enable epoch-based timeout
               let engine = Engine::new(&config)?;
               Ok(Self { engine })
           }
           
           pub fn engine(&self) -> &Engine { &self.engine }
       }
       ```

    2. **`WasiContextBuilder`** — custom WASI context with capability control:
       ```rust
       fn build_wasi_ctx(wasm_config: &WasmConfig, skill_name: &str) -> Result<WasiCtx> {
           let data_dir = skill_data_dir(skill_name);
           std::fs::create_dir_all(&data_dir)?;
           
           let mut builder = WasiCtxBuilder::new();
           
           // Filesystem: bind only the skill's data directory for writes
           // (if permissions allow)
           if has_write_permission(wasm_config, &data_dir) {
               builder = builder.preopened_dir(
                   &data_dir,
                   "/data",           // WASM-visible path
                   DirPerms::READ_WRITE,
                   FilePerms::WRITE,
               )?;
           }
           // Always allow read from /data
           builder = builder.preopened_dir(
               &data_dir,
               "/data",
               DirPerms::READ,
               FilePerms::READ,
           )?;
           
           // No environment variables
           builder = builder.inherit_stderr();  // Logging
           
           Ok(builder.build())
       }
       ```

       Note: wasmtime-wasi's `WasiCtxBuilder` API may differ between versions. Check the actual API on docs.rs for the version used. The key constraint: NO network sockets, NO env vars, NO stdin, stderr-only for logging.

    **Step B — Implement WASI capability gating in `wasm.rs`**:
    
    The `build_wasi_ctx` function must enforce:
    1. **No network**: Do NOT add `tcp`, `udp`, or `http` WASI capabilities. If wasmtime-wasi's `WasiCtxBuilder` has methods for network sockets, simply don't call them.
    2. **No environment**: Do NOT call `env()` or `envs()` on the builder. Call `inherit_stderr()` for logging only.
    3. **No random**: Do NOT add random capabilities unless explicitly needed.
    4. **Restricted filesystem**: Only preopen the skill's data dir. Do NOT preopen any system paths.
    
    The exact `WasiCtxBuilder` API depends on the wasmtime version. Check the crate docs:
    - wasmtime 29.x uses WASI preview 2 API: `WasiCtxBuilder::new()`, `.preopened_dir()`, `.inherit_stderr()`, `.build()`
    - If preview 2 API differs, adapt accordingly.
    - NO: `.env()`, `.inherit_stdin()`, `.inherit_stdout()`, `.socket()`, `.tcp()`
    
    **Step C — Write Tests** (`wasm_tests.rs`, `#[cfg(test)] mod tests`):
    
    Create test WAT modules inline (using `wat::parse_str` from [dev-dependencies]):
