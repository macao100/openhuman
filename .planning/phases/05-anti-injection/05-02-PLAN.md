<objective>
Wrap all WASM skill outputs in a structured JSON envelope before they reach the LLM conversation context, preventing raw skill output from being interpreted as instructions.

Purpose: When a WASM skill returns text, that text could contain prompt injection payloads. By encapsulating it in a typed JSON envelope that the system prompt's anti-injection section (Plan 05-01) explains, the LLM sees structured data rather than raw text that could be mistaken for instructions.

Output:
- `SkillOutputEnvelope` struct (JSON-serializable)
- `execute_wasm_structured()` function that wraps raw WASM output
- Tool loop integration that uses structured output for skill tools
- Tests verifying JSON envelope format and LLM-facing presentation
</objective>

       ```rust
       use serde::Serialize;

       /// Structured JSON envelope for WASM skill outputs.
       /// Prevents raw skill output from reaching the LLM prompt as plain text.
       #[derive(Debug, Clone, Serialize)]
       pub struct SkillOutputEnvelope {
           /// Name of the skill that produced this output (from manifest).
           pub skill_name: String,
           /// Version of the skill that produced this output.
           pub version: String,
           /// MIME-like content type hint (e.g. "text/plain", "application/json").
           pub content_type: String,
           /// Always false for skill output in v1 (all skill output is untrusted).
           pub trusted: bool,
           /// The actual output content, truncated if necessary.
           pub output: String,
           /// Whether the output was truncated to fit max_chars.
           pub truncated: bool,
           /// Number of characters in the original (pre-truncation) output.
           #[serde(default)]
           pub original_length: usize,
       }

       impl SkillOutputEnvelope {
           /// Create a new envelope from raw skill output bytes.
           ///
           /// The output is decoded as UTF-8 (lossy), truncated to `max_chars`
           /// if specified, and wrapped in the envelope struct.
           pub fn new(
               skill_name: impl Into<String>,
               version: impl Into<String>,
               raw_output: &[u8],
               max_chars: Option<usize>,
           ) -> Self {
               let output = String::from_utf8_lossy(raw_output).to_string();
               let original_length = output.chars().count();
               let max_chars = max_chars.unwrap_or(100_000); // generous default
               let (truncated, final_output) = if original_length > max_chars {
                   let truncated: String = output.chars().take(max_chars).collect();
                   (true, truncated)
               } else {
                   (false, output)
               };
               Self {
                   skill_name: skill_name.into(),
                   version: version.into(),
                   content_type: "text/plain".to_string(),
                   trusted: false,
                   output: final_output,
                   truncated,
                   original_length,
               }
           }

           /// Render the envelope as a single compact JSON line.
           pub fn to_json_line(&self) -> String {
               serde_json::to_string(self).unwrap_or_else(|_| {
                   format!("{{\"error\":\"serialization failed\",\"skill_name\":\"{}\"}}", self.skill_name)
               })
           }
       }
       ```
