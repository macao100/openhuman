    ```rust
    use serde::{Deserialize, Serialize};

    /// Verdict rendu par le LLM N3.
    #[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
    #[serde(rename_all = "lowercase")]
    pub enum N3Verdict {
        Allow,
        Block,
        Uncertain,
    }

    /// Resultat de l'evaluation N3.
    #[derive(Debug, Clone, Serialize, Deserialize)]
    pub struct N3Result {
        pub verdict: N3Verdict,
        pub reason: String,
        pub latency_us: u64,
        pub cached: bool,
        pub model_used: String,
    }

    impl N3Result {
        /// Fail-closed: Block et Uncertain sont consideres comme bloquants.
        pub fn should_block(&self) -> bool {
            matches!(self.verdict, N3Verdict::Block | N3Verdict::Uncertain)
        }

        /// Parse la reponse JSON du LLM en N3Verdict + reason.
        pub fn from_llm_response(response: &str) -> Option<Self> {
            // Cherche un objet JSON dans la reponse
            // Pattern: {"verdict": "allow", "reason": "..."}
            let start = response.find('{')?;
            let end = response.rfind('}')?;
            let json_str = &response[start..=end];
            let parsed: LlmResponse = serde_json::from_str(json_str).ok()?;
            Some(N3Result {
                verdict: parsed.verdict,
                reason: parsed.reason,
                latency_us: 0,
                cached: false,
                model_used: String::new(),
            })
        }
    }

    /// Format de reponse attendu du LLM.
    #[derive(Debug, Deserialize)]
    struct LlmResponse {
        verdict: N3Verdict,
        #[serde(default)]
        reason: String,
    }

    /// Configuration N3.
    pub struct N3Config {
        pub enabled: bool,           // defaut: true
        pub max_tokens: u32,         // defaut: 256
        pub timeout_ms: u64,         // defaut: 450 (<500ms target avec marge)
        pub cache_size: usize,       // defaut: 100
        pub model_override: Option<String>, // None = utiliser le modele par defaut
    }
    ```

    ```rust
    use std::collections::HashMap;

    /// Cache LRU simple pour les resultats de validation N3.
    /// Taille max configurable. Thread-safe via Mutex externe.
    pub struct LruCache<V> {
        max_size: usize,
        entries: HashMap<String, CacheEntry<V>>,
        order: Vec<String>,  // Plus recent en dernier
    }

    struct CacheEntry<V> {
        value: V,
    }

    impl<V: Clone> LruCache<V> {
        pub fn new(max_size: usize) -> Self;
        pub fn get(&mut self, key: &str) -> Option<V>;
        pub fn insert(&mut self, key: String, value: V);
    }
    ```

    Completer `n3/mod.rs` avec la logique d'appel LLM dans `evaluate()`:

    1. Generer la cache_key depuis les parametres
    2. Verifier le cache (si present et que le timeout N2 n'a pas change -> retourner)
    3. Si cache miss:
       a. Construire le prompt via `N3PromptBuilder`
       b. Charger la config (`crate::openhuman::config::load_config()`)
       c. Appeler `local_ai_prompt(&config, &full_prompt, Some(config.max_tokens), Some(true)).await`
       d. Timeout avec `tokio::time::timeout(Duration::from_millis(config.timeout_ms), ...)`
       e. Parser la reponse JSON via `N3Result::from_llm_response()`
       f. Si parsing echoue -> retourner N3Result avec verdict Uncertain + raison "LLM response parse failed"
       g. Si timeout -> retourner N3Result avec verdict Uncertain + raison "N3 validation timed out"
       h. Stocker dans le cache
    4. Retourner le N3Result

    Le timeout est critique: si le LLM local prend trop de temps, on retourne Uncertain (fail-closed bloque l'action). Ceci respecte le target <500ms.

    ```rust
    use tokio::time::timeout;
    use std::time::{Duration, Instant};

    pub async fn evaluate(&self, ...) -> N3Result {
        let start = Instant::now();
        let key = N3PromptBuilder::cache_key(tool_name, tool_args, command);

        // Cache check
        {
            let mut cache = self.cache.lock();
            if let Some(cached) = cache.get(&key) {
                let mut result = cached.clone();
                result.cached = true;
                return result;
            }
        }

        // Build prompt
        let full_prompt = format!(
            "{}\n\n{}",
            N3PromptBuilder::system_prompt(),
            N3PromptBuilder::build_user_prompt(tool_name, tool_args, command, file_path, n2_scores)
        );

        // LLM call with timeout
        let llm_future = Self::call_llm(&full_prompt, self.config.max_tokens);
        let llm_result = tokio::time::timeout(
            Duration::from_millis(self.config.timeout_ms),
            llm_future,
        ).await;

        let latency_us = start.elapsed().as_micros() as u64;

        let mut n3_result = match llm_result {
            Ok(Ok(response)) => {
                N3Result::from_llm_response(&response).unwrap_or(N3Result {
                    verdict: N3Verdict::Uncertain,
                    reason: "Failed to parse LLM response".into(),
                    latency_us,
                    cached: false,
                    model_used: "local".into(),
                })
            }
            Ok(Err(e)) => N3Result {
                verdict: N3Verdict::Uncertain,
                reason: format!("LLM error: {}", e),
                latency_us,
                cached: false,
                model_used: "local".into(),
            },
            Err(_) => N3Result {
                verdict: N3Verdict::Uncertain,
                reason: "N3 validation timed out (>{}ms)".into(),
                latency_us,
                cached: false,
                model_used: "local".into(),
            },
        };

        n3_result.latency_us = latency_us;

        // Cache insert
        {
            let mut cache = self.cache.lock();
            cache.insert(key, n3_result.clone());
        }

        n3_result
    }

    async fn call_llm(prompt: &str, max_tokens: u32) -> Result<String, String> {
        let config = crate::openhuman::config::load_config()
            .map_err(|e| format!("config load failed: {}", e))?;
        let result = crate::openhuman::inference::local::ops::local_ai_prompt(
            &config, prompt, Some(max_tokens), Some(true)
        ).await?;
        Ok(result.value)
    }
    ```
